import cv2

img = cv2.imread("C:\Program Files\OpenCV\img1.png")
(h, w) = img.shape[:2]
center = (w//2, h//2)

clockwise = cv2.getRotationMatrix2D(center, -45, 1.0)
anticlockwise = cv2.getRotationMatrix2D(center, 45, 1.0)

rot1 = cv2.warpAffine(img, clockwise, (w, h))
rot2 = cv2.warpAffine(img, anticlockwise, (w, h))

cv2.imshow("Original", img)
cv2.imshow("Clockwise Rotation", rot1)
cv2.imshow("Anti-Clockwise Rotation", rot2)

cv2.waitKey(0)
cv2.destroyAllWindows()
