import cv2
import numpy as np

img = cv2.imread("C:\Program Files\OpenCV\img1.png")

pts1 = np.float32([[50,50], [300,50], [50,300], [300,300]])
pts2 = np.float32([[10,100], [280,20], [100,320], [320,300]])

H, status = cv2.findHomography(pts1, pts2)
output = cv2.warpPerspective(img, H, (400, 400))

cv2.imshow("Original Image", img)
cv2.imshow("Homography Output", output)

cv2.waitKey(0)
cv2.destroyAllWindows()
